FORM get_data .
  CLEAR : gt_data, gt_data[].

* MARA : a,  MAKT : b
  SELECT a~matnr b~maktx a~matkl a~mtart a~meins
    INTO CORRESPONDING FIELDS OF TABLE gt_data
    FROM mara AS a INNER JOIN makt AS b
                           ON a~matnr = b~matnr
                          AND b~spras = p_spras
                        WHERE a~mtart IN s_mtart
                          AND a~matkl IN s_matkl
                          AND a~matnr IN s_matnr.
ENDFORM. " GET_DATA

FORM display_screen .

  IF go_container IS NOT BOUND.

    PERFORM create_alv_grid.

    PERFORM set_fcat.
    PERFORM set_layout.
    PERFORM set_etc.

    PERFORM display_alv_grid.

  ELSE.
    PERFORM refresh_alv.
  ENDIF.

ENDFORM. " DISPLAY_SCREEN
